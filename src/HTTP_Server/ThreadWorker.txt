    static ReadWriteLock rwLock = new ReentrantReadWriteLock();
    String fName = null;

    public ThreadWorker(String fName) {
        this.fName = fName;
    }

    public void write() {
        rwLock.writeLock().lock();
        try {
            //write to store;
        } catch (Exception e) {
            //Handle Exception here ...
        } finally {
            rwLock.writeLock().unlock();
        }
    }

    public String read() {
        rwLock.readLock().lock();
        try {
            //read from store;
        } catch (Exception e) {
            //Handle Exception here ...
        }
        finally {
            rwLock.readLock().unlock();
        }
        return null;
    }
}
